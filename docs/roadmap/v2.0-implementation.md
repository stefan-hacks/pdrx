# PDRX Feature Roadmap: Evolution to Debian State Engine

> **Vision:** Transform pdrx from a package manager wrapper into a complete system state capture and restoration engine for Debian-based distributions.

## Executive Summary

This roadmap outlines the evolution of pdrx from its current "imperative package management with auto-config sync" to a comprehensive **Debian State Engine** that captures and reproduces entire system configurations. The goal is achieving 90-95% of NixOS-style reproducibility while maintaining Debian's familiarity and compatibility.

---

## Current State Analysis

### ✅ What PDRX Does Well Today

| Feature | Status | Notes |
|---------|--------|-------|
| Package tracking (apt, dnf, pacman, etc.) | ✅ Stable | Core differentiator with PM tagging |
| Auto-config sync on install/remove | ✅ Stable | No manual YAML editing required |
| Cross-distro support | ✅ Stable | Works on 8+ package managers |
| Basic dotfile tracking | ✅ Stable | Native implementation |
| Desktop settings export/import | ✅ Stable | GNOME, KDE, i3 support |
| Version pinning | ✅ Stable | Per-package version locks |
| Backup generations | ✅ Stable | Rollback capability |

### ❌ Critical Gaps (The Opportunity)

| Gap | Impact | User Pain Point |
|-----|--------|-----------------|
| No auto-discovery of existing state | **HIGH** | Manual sync after "normal" package installs |
| No GNOME extensions management | **HIGH** | Extensions must be manually reinstalled |
| No Flatpak remote tracking | **MEDIUM** | Remotes lost on new machine |
| No APT repository capture | **MEDIUM** | Third-party repos must be manually re-added |
| No systemd service state | **MEDIUM** | Services don't auto-enable |
| No container/VM snapshots | **LOW** | Dev environments must be rebuilt |
| No secrets management | **LOW** | Manual credential handling |

**Goal:** Transform sync from "dump package list" to "comprehensive system state capture."

#### Base Plugin Interface

```python
# pdrx/plugins/base.py
from abc import ABC, abstractmethod
from typing import Dict, List, Any
from dataclasses import dataclass

@dataclass
class DiscoveryResult:
    plugin_name: str
    data: Dict[str, Any]
    warnings: List[str]
    can_restore: bool  # Does this plugin support restoration?

class DiscoveryPlugin(ABC):
    name: str
    description: str
    
    @abstractmethod
    def discover(self) -> DiscoveryResult:
        """Scan system and return current state."""
        pass
    
    @abstractmethod
    def can_restore(self) -> bool:
        """Whether this plugin can restore state."""
        pass
    
    @abstractmethod
    def restore(self, data: Dict[str, Any]) -> bool:
        """Apply configuration from saved state."""
        pass
    
    def validate(self, data: Dict[str, Any]) -> List[str]:
        """Validate configuration before restore."""
        return []
```

### 1.2 Priority: Flatpak State (P0)

**Rationale:** Low complexity, high user value, existing user base.

```python
# pdrx/plugins/flatpaks.py
import subprocess
import json
from typing import Dict, List

class FlatpakPlugin(DiscoveryPlugin):
    name = "flatpak"
    description = "Flatpak applications and remotes"
    
    def discover(self) -> DiscoveryResult:
        apps = self._discover_apps()
        remotes = self._discover_remotes()
        
        return DiscoveryResult(
            plugin_name=self.name,
            data={
                "remotes": remotes,
                "apps": apps
            },
            warnings=[],
            can_restore=True
        )
    
    def _discover_apps(self) -> List[Dict]:
        """Discover installed Flatpak apps."""
        cmd = ["flatpak", "list", "--app", "--columns=application,origin,installation,version"]
        result = subprocess.run(cmd, capture_output=True, text=True)
        
        apps = []
        for line in result.stdout.strip().split('\n'):
            if '\t' in line:
                parts = line.split('\t')
                apps.append({
                    "id": parts[0],
                    "remote": parts[1],
                    "installation": parts[2],  # system or user
                    "version": parts[3] if len(parts) > 3 else None
                })
        return apps
    
    def _discover_remotes(self) -> List[Dict]:
        """Discover configured Flatpak remotes."""
        cmd = ["flatpak", "remotes", "--columns=name,url,options"]
        result = subprocess.run(cmd, capture_output=True, text=True)
        
        remotes = []
        for line in result.stdout.strip().split('\n'):
            if '\t' in line:
                parts = line.split('\t')
                remotes.append({
                    "name": parts[0],
                    "url": parts[1],
                    "options": parts[2].split(',') if len(parts) > 2 else []
                })
        return remotes
    
    def restore(self, data: Dict) -> bool:
        # 1. Add remotes first
        for remote in data.get("remotes", []):
            cmd = ["flatpak", "remote-add", "--if-not-exists"]
            if "user" in remote.get("options", []):
                cmd.append("--user")
            cmd.extend([remote["name"], remote["url"]])
            subprocess.run(cmd, check=True)
        
        # 2. Install apps
        for app in data.get("apps", []):
            cmd = ["flatpak", "install", "-y"]
            if app.get("installation") == "user":
                cmd.append("--user")
            cmd.append(app["remote"])
            cmd.append(app["id"])
            subprocess.run(cmd, check=True)
        
        return True
```

### 1.3 Priority: GNOME Extensions (P0)

**Rationale:** Major pain point, no existing solution.

```python
# pdrx/plugins/gnome_extensions.py
import subprocess
import json
import os
from pathlib import Path

class GnomeExtensionsPlugin(DiscoveryPlugin):
    name = "gnome_extensions"
    description = "GNOME Shell extensions with settings"
    
    EXTENSIONS_DIR = Path.home() / ".local/share/gnome-shell/extensions"
    
    def discover(self) -> DiscoveryResult:
        extensions = self._discover_enabled_extensions()
        warnings = []
        
        # Warn about system extensions that can't be auto-installed
        system_exts = self._discover_system_extensions()
        if system_exts:
            warnings.append(f"Found {len(system_exts)} system extensions that may require manual installation")
        
        return DiscoveryResult(
            plugin_name=self.name,
            data={"extensions": extensions},
            warnings=warnings,
            can_restore=True
        )
    
    def _discover_enabled_extensions(self) -> List[Dict]:
        """Discover user-installed extensions."""
        cmd = ["gsettings", "get", "org.gnome.shell", "enabled-extensions"]
        result = subprocess.run(cmd, capture_output=True, text=True)
        
        extensions = []
        enabled_uuids = eval(result.stdout.strip())  # Returns Python list syntax
        
        for uuid in enabled_uuids:
            ext_info = self._get_extension_info(uuid)
            if ext_info:
                extensions.append(ext_info)
        
        return extensions
    
    def _get_extension_info(self, uuid: str) -> Dict:
        """Get detailed info about an extension."""
        ext_path = self.EXTENSIONS_DIR / uuid
        if not ext_path.exists():
            return None  # System extension
        
        metadata_path = ext_path / "metadata.json"
        if not metadata_path.exists():
            return {"uuid": uuid, "version": "unknown"}
        
        with open(metadata_path) as f:
            metadata = json.load(f)
        
        # Get extension settings from dconf
        settings = self._get_extension_settings(uuid)
        
        return {
            "uuid": uuid,
            "name": metadata.get("name", uuid),
            "version": metadata.get("version", "unknown"),
            "download_url": metadata.get("url", None),
            "settings": settings
        }
    
    def _get_extension_settings(self, uuid: str) -> Dict:
        """Get dconf settings for an extension."""
        schema = f"org.gnome.shell.extensions.{uuid.split('@')[0].replace('-', '_')}"
        
        cmd = ["gsettings", "list-recursively", schema]
        result = subprocess.run(cmd, capture_output=True, text=True)
        
        settings = {}
        if result.returncode == 0:
            for line in result.stdout.strip().split('\n'):
                if ' ' in line:
                    parts = line.split(' ', 2)
                    if len(parts) >= 3:
                        key = parts[1]
                        value = parts[2]
                        settings[key] = value
        
        return settings
    
    def restore(self, data: Dict) -> bool:
        """Install extensions from extensions.gnome.org."""
        for ext in data.get("extensions", []):
            # Method 1: If metadata has download URL, fetch directly
            # Method 2: Use gnome-extensions CLI if available
            # Method 3: Guide user to browser installation
            
            uuid = ext["uuid"]
            version = ext.get("version")
            
            # Install via gnome-extensions if available
            result = subprocess.run(
                ["gnome-extensions", "install", uuid],
                capture_output=True
            )
            
            if result.returncode != 0:
                # Fall back to manual instructions
                print(f"Please install extension {uuid} from extensions.gnome.org")
                print(f"  Version: {version}")
            
            # Restore settings via dconf
            settings = ext.get("settings", {})
            schema = f"org.gnome.shell.extensions.{uuid.split('@')[0].replace('-', '_')}"
            
            for key, value in settings.items():
                subprocess.run(
                    ["gsettings", "set", f"{schema}.{key}", value],
                    capture_output=True
                )
        
        return True
```

**Challenge:** GNOME extensions can't be fully automated due to browser-based installation. Solutions:
1. Integrate with `chrome-gnome-shell` connector
2. Cache extension ZIPs locally
3. Provide clear manual instructions

### 2.3 Config Loader Implementation

```python
# pdrx/config/loader.py
import yaml
import json
from pathlib import Path
from typing import Dict, Any, List, Optional
import jsonschema
from dataclasses import dataclass

@dataclass
class Config:
    version: str
    data: Dict[str, Any]
    profile: Optional[str]
    base_path: Path

class ConfigLoader:
    SCHEMA_PATH = Path(__file__).parent / "schemas" / "config-v2.0.json"
    
    def __init__(self, config_dir: Path = None):
        self.config_dir = config_dir or Path.home() / ".pdrx/config"
        self.profiles_dir = self.config_dir.parent / "profiles"
        
        with open(self.SCHEMA_PATH) as f:
            self.schema = json.load(f)
    
    def load(self, profile: str = None) -> Config:
        """Load configuration with inheritance resolution."""
        config_path = self._get_config_path(profile)
        
        if not config_path.exists():
            raise FileNotFoundError(f"Config not found: {config_path}")
        
        with open(config_path) as f:
            data = yaml.safe_load(f)
        
        # Resolve inheritance
        data = self._resolve_inheritance(data, profile)
        
        # Validate against schema
        self._validate(data)
        
        return Config(
            version=data["pdrx"]["version"],
            data=data,
            profile=profile or data.get("pdrx", {}).get("profile"),
            base_path=config_path.parent
        )
    
    def _resolve_inheritance(self, data: Dict, profile: str) -> Dict:
        """Resolve profile inheritance chain."""
        inherits = data.get("inherits", [])
        
        if not inherits:
            return data
        
        merged = {}
        
        # Load base profiles first (left to right)
        for base_profile in inherits:
            base_path = self.profiles_dir / f"{base_profile}.yaml"
            if base_path.exists():
                with open(base_path) as f:
                    base_data = yaml.safe_load(f)
                merged = self._deep_merge(merged, base_data)
        
        # Overlay current profile (rightmost wins)
        merged = self._deep_merge(merged, data)
        
        return merged
    
    def _deep_merge(self, base: Dict, overlay: Dict) -> Dict:
        """Deep merge two dictionaries."""
        result = base.copy()
        
        for key, value in overlay.items():
            if key in result and isinstance(result[key], dict) and isinstance(value, dict):
                result[key] = self._deep_merge(result[key], value)
            elif key in result and isinstance(result[key], list) and isinstance(value, list):
                # For lists, overlay replaces (useful for packages)
                result[key] = value
            else:
                result[key] = value
        
        return result
    
    def _validate(self, data: Dict):
        """Validate configuration against JSON schema."""
        try:
            jsonschema.validate(data, self.schema)
        except jsonschema.ValidationError as e:
            raise ValueError(f"Config validation error: {e.message} at {'.'.join(str(p) for p in e.path)}")
    
    def _get_config_path(self, profile: str = None) -> Path:
        """Get configuration file path."""
        if profile:
            return self.profiles_dir / f"{profile}.yaml"
        return self.config_dir / "machine.yaml"
```

# pdrx/plugins/apt_repos.py
import subprocess
import re
from pathlib import Path
from typing import Dict, List

class AptReposPlugin(DiscoveryPlugin):
    name = "apt_repos"
    description = "APT repository sources and GPG keys"
    
    SOURCES_DIR = Path("/etc/apt/sources.list.d")
    KEYRINGS_DIR = Path("/etc/apt/keyrings")
    
    def discover(self) -> DiscoveryResult:
        repos = self._discover_sources()
        keys = self._discover_keys()
        
        return DiscoveryResult(
            plugin_name=self.name,
            data={
                "repos": repos,
                "keys": keys
            },
            warnings=[],
            can_restore=True
        )
    
    def _discover_sources(self) -> List[Dict]:
        """Discover third-party APT sources."""
        repos = []
        
        for source_file in self.SOURCES_DIR.glob("*.list"):
            # Skip standard Debian repos
            if self._is_standard_repo(source_file):
                continue
            
            with open(source_file) as f:
                for line in f:
                    line = line.strip()
                    if line.startswith('#') or not line:
                        continue
                    
                    repo = self._parse_sources_line(line)
                    if repo:
                        repo['file'] = source_file.name
                        repos.append(repo)
        
        return repos
    
    def _is_standard_repo(self, path: Path) -> bool:
        """Check if this is a standard Debian repo file."""
        standard_files = ['debian.sources', 'debian.list']
        return path.name in standard_files
    
    def _parse_sources_line(self, line: str) -> Dict:
        """Parse a sources.list line into components."""
        # Format: deb [arch=amd64 signed-by=/path] https://url suite component
        pattern = r'^(deb|deb-src)\s+(?:\[([^\]]+)\]\s+)?(\S+)\s+(\S+)\s+(.*)$'
        match = re.match(pattern, line)
        
        if not match:
            return None
        
        repo_type, options, uri, suite, components = match.groups()
        
        # Parse options
        opts = {}
        if options:
            for opt in options.split():
                if '=' in opt:
                    k, v = opt.split('=', 1)
                    opts[k] = v
        
        return {
            "type": repo_type,
            "uri": uri,
            "suite": suite,
            "components": components.split(),
            "arch": opts.get('arch', 'amd64'),
            "signed_by": opts.get('signed-by', ''),
        }
    
    def _discover_keys(self) -> List[Dict]:
        """Discover GPG keys in keyrings directory."""
        keys = []
        
        for key_file in self.KEYRINGS_DIR.glob("*.gpg"):
            keys.append({
                "name": key_file.stem,
                "path": str(key_file),
                "type": "gpg"
            })
        
        for key_file in self.KEYRINGS_DIR.glob("*.asc"):
            keys.append({
                "name": key_file.stem,
                "path": str(key_file),
                "type": "asc"
            })
        
        return keys
    
    def restore(self, data: Dict) -> bool:
        """Add APT repositories and keys."""
        # Add GPG keys first
        for key in data.get("keys", []):
            self._add_key(key)
        
        # Add sources
        for repo in data.get("repos", []):
            self._add_repo(repo)
        
        # Update package list
        subprocess.run(["apt-get", "update"], check=True)
        
        return True
    
    def _add_key(self, key: Dict):
        """Add a GPG key to keyrings."""
        if key.get("url"):
            # Download from URL
            dest = self.KEYRINGS_DIR / f"{key['name']}.asc"
            subprocess.run([
                "curl", "-fsSL", key["url"],
                "-o", str(dest)
            ], check=True)
        elif key.get("content"):
            # Write from embedded content
            dest = self.KEYRINGS_DIR / f"{key['name']}.asc"
            with open(dest, 'w') as f:
                f.write(key["content"])
    
    def _add_repo(self, repo: Dict):
        """Add an APT source."""
        # Modern deb822 format (preferred)
        if repo.get("format") == "deb822":
            content = self._format_deb822(repo)
        else:
            content = self._format_traditional(repo)
        
        filename = repo.get("filename", f"{repo['name']}.sources")
        dest = self.SOURCES_DIR / filename
        
        with open(dest, 'w') as f:
            f.write(content)
    
    def _format_deb822(self, repo: Dict) -> str:
        """Format repository in deb822 format."""
        lines = [
            "Types: deb",
            f"URIs: {repo['uri']}",
            f"Suites: {repo['suite']}",
            f"Components: {' '.join(repo['components'])}",
            f"Architectures: {repo.get('arch', 'amd64')}",
        ]
        
        if repo.get("signed_by"):
            lines.append(f"Signed-By: {repo['signed_by']}")
        
        return '\n'.join(lines)
```

# pdrx/plugins/systemd_services.py
import subprocess
from typing import Dict, List

class SystemdServicesPlugin(DiscoveryPlugin):
    name = "systemd_services"
    description = "Systemd enabled/disabled services"
    
    def discover(self) -> DiscoveryResult:
        services = self._discover_services()
        
        return DiscoveryResult(
            plugin_name=self.name,
            data=services,
            warnings=[],
            can_restore=True
        )
    
    def _discover_services(self) -> Dict[str, List[str]]:
        """Discover service states."""
        result = {
            "enabled": [],
            "disabled": [],
            "masked": []
        }
        
        # Get enabled services
        enabled = subprocess.run(
            ["systemctl", "list-unit-files", "--state=enabled", "--type=service", "--no-legend", "--quiet"],
            capture_output=True, text=True
        )
        
        for line in enabled.stdout.strip().split('\n'):
            if line:
                service_name = line.split()[0].replace('.service', '')
                # Filter out auto-enabled system services
                if not self._is_auto_system_service(service_name):
                    result["enabled"].append(service_name)
        
        # Get masked services (user explicitly disabled)
        masked = subprocess.run(
            ["systemctl", "list-unit-files", "--state=masked", "--type=service", "--no-legend", "--quiet"],
            capture_output=True, text=True
        )
        
        for line in masked.stdout.strip().split('\n'):
            if line:
                service_name = line.split()[0].replace('.service', '')
                result["masked"].append(service_name)
        
        return result
    
    def _is_auto_system_service(self, name: str) -> bool:
        """Check if this is an auto-enabled system service."""
        auto_services = [
            "apparmor", "cron", "dbus", "networking", "rsyslog",
            "systemd-timesyncd", "unattended-upgrades"
        ]
        return name in auto_services
    
    def restore(self, data: Dict) -> bool:
        """Enable/disable services according to config."""
        # Enable services
        for service in data.get("enabled", []):
            subprocess.run(
                ["systemctl", "enable", "--now", f"{service}.service"],
                capture_output=True
            )
        
        # Mask explicitly disabled
        for service in data.get("masked", []):
            subprocess.run(
                ["systemctl", "mask", f"{service}.service"],
                capture_output=True
            )
        
        return True
```

# pdrx/commands/doctor.py
import subprocess
from typing import List, Dict

class HealthCheck:
    """Verify system health against configuration."""
    
    def run(self) -> List[Dict]:
        checks = [
            self._check_apt_repos(),
            self._check_gpg_keys(),
            self._check_packages(),
            self._check_services(),
            self._check_gnome_extensions(),
        ]
        return [c for c in checks if c]  # Filter out None
    
    def _check_apt_repos(self) -> Dict:
        """Verify APT repos are reachable."""
        issues = []
        
        result = subprocess.run(
            ["apt-get", "update", "-qq"],
            capture_output=True
        )
        
        if result.returncode != 0:
            issues.append("APT update failed - check repository URLs")
        
        return {
            "category": "APT Repositories",
            "status": "pass" if not issues else "fail",
            "issues": issues
        }
    
    def _check_gnome_extensions(self) -> Dict:
        """Verify GNOME extensions are compatible."""
        issues = []
        
        # Get current GNOME Shell version
        result = subprocess.run(
            ["gnome-shell", "--version"],
            capture_output=True, text=True
        )
        
        if result.returncode == 0:
            version = result.stdout.strip().split()[-1]
            
            # Check each configured extension
            for ext in self.config.get("desktop", {}).get("gnome", {}).get("extensions", []):
                if not self._is_extension_compatible(ext, version):
                    issues.append(f"Extension {ext['uuid']} may not be compatible with GNOME {version}")
        
        return {
            "category": "GNOME Extensions",
            "status": "pass" if not issues else "warn",
            "issues": issues
        }
